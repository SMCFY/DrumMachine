135function checkForTampering(plugin, paramState, sampleRate, cause)
136% Verify parameters were not tampered with
137if ~isequal(paramState.strikePos, plugin.strikePos)
138    error('ValidateAudioPlugin:ParamChanged', ...
139        '%s changed parameter ''strikePos'' from %g to %g.', ...
140        cause, paramState.strikePos, plugin.strikePos);
141end
142if ~isequal(paramState.strikeVig, plugin.strikeVig)
143    error('ValidateAudioPlugin:ParamChanged', ...
144        '%s changed parameter ''strikeVig'' from %g to %g.', ...
145        cause, paramState.strikeVig, plugin.strikeVig);
146end
147if ~isequal(paramState.dimension, plugin.dimension)
148    error('ValidateAudioPlugin:ParamChanged', ...
149        '%s changed parameter ''dimension'' from %g to %g.', ...
150        cause, paramState.dimension, plugin.dimension);
151end
152if ~isequal(paramState.material, plugin.material)
153    error('ValidateAudioPlugin:ParamChanged', ...
154        '%s changed parameter ''material'' from %g to %g.', ...
155        cause, paramState.material, plugin.material);
156end
157if ~isequal(paramState.instID, plugin.instID)
158    error('ValidateAudioPlugin:ParamChanged', ...
159        '%s changed parameter ''instID'' from %g to %g.', ...
160        cause, paramState.instID, plugin.instID);
161end
162if ~isequal(paramState.trig, plugin.trig)
163    error('ValidateAudioPlugin:ParamChanged', ...
164        '%s changed parameter ''trig'' from ''%s'' to ''%s'.'', ...
165        cause, paramState.trig, plugin.trig);
166end
167% Verify sample rate was not tampered with
168if ~isequal(getSampleRate(plugin), sampleRate)
169    error('ValidateAudioPlugin:SampleRateChanged', ...
170        '%s changed sample rate from %g to %g.', ...
171        cause, sampleRate, getSampleRate(plugin));
172end
173end

175function y = logchirp(f0, f1, Fs, nsamples, initialPhase)
176% logarithmically swept sine from f0 to f1 over nsamples, at Fs
177y = zeros(nsamples,numel(initialPhase));
178instPhi = logInstantaneousPhase(f0, f1, Fs, nsamples);
179for i = 1:numel(initialPhase)
180    y(:,i) = sin(instPhi + initialPhase(i));
181end
182end
183
184function phi = logInstantaneousPhase(f0, f1, Fs, n)
185final = n-1;
186t = (0:final)/final;
187t1 = final/Fs;
188phi = 2*pi * t1/log(f1/f0) * (f0 * (f1/f0).^(t') - f0);
189end

191function paramState = initParamState(plugin)
192paramState.strikePos = plugin.strikePos;
193paramState.strikeVig = plugin.strikeVig;
194paramState.dimension = plugin.dimension;
195paramState.material = plugin.material;
196paramState.instID = plugin.instID;
197paramState.trig = plugin.trig;
198end

200function val = fromNormalizedstrikePos(normval)
201val = 0 + (1-0)*normval;
202end
203
204function val = fromNormalizedstrikeVig(normval)
205val = 0 + (1-0)*normval;
206end
207
208function val = fromNormalizeddimension(normval)
209val = 0.5 + (1.5-0.5)*normval;
210end
211
212function val = fromNormalizedmaterial(normval)
213val = 0 + (1-0)*normval;
214end
215
216function val = fromNormalizedinstID(normval)
217val = floor(0.5 + 1 + (4-1)*normval);
218end
219
220function val = fromNormalizedtrig(normval)
221idx = floor( 0.5 + normval * 1);
222switch idx
223    case 0
224        val = 'noteOff';
225    otherwise
226        val = 'noteOn_';
227end
228end
